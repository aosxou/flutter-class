# 🧩 Pattern Printer (Dart)
다양한 문자 패턴(#, space)을 반복문으로 출력.

---

## 개요
이 프로그램은 세 가지 도형 패턴을 콘솔에 출력.
1. 정사각형 패턴  
2. 대각선 패턴  
3. X자 패턴  

---

## 💻 코드

```dart
// pattern_printer.dart
// 다양한 문자 패턴(#, space)을 반복문으로 출력하는 예제
// 작성자: 이가윤
// 설명: 여러 형태의 도형 패턴을 콘솔에 출력하는 프로그램

void main() {
  var size = 5;

  print("=== 패턴 1: # 정사각형 ===");
  for (var i = 0; i < size; i++) {
    printSharp(size, i);
  }

  print("\n=== 패턴 2: 대각선 ===");
  for (var i = 0; i < size; i++) {
    print2(size, i);
  }

  print("\n=== 패턴 3: X자 모양 ===");
  for (var i = 0; i < size; i++) {
    print1(size, i);
  }

  print("\n=== 모든 패턴 출력 완료 ===");
}

/// ■ 패턴 1: # 정사각형 패턴
/// - 윗줄, 아랫줄은 전부 #
/// - 가운데 줄은 #   # 구조 (중앙은 비움)
void printSharp(int size, int y) {
  var result = "";

  for (var x = 0; x < size; x++) {
    bool condition;

    if (y == 0 || y == size - 1) {
      // 첫 줄과 마지막 줄은 전부 #
      condition = true;
    } else if (y == size ~/ 2) {
      // 가운데 줄은 양쪽과 중앙만 #
      condition = (x == 0 || x == size ~/ 2 || x == size - 1);
    } else {
      // 나머지 줄은 양쪽만 #
      condition = (x <= 1 || x >= size - 2);
    }

    result += condition ? "#" : " ";
  }
  print(result);
}

/// ■ 패턴 2: 대각선 (\ 방향)
/// - 왼쪽 위에서 오른쪽 아래로 #
void print2(int size, int y) {
  var result = "";
  for (var x = 0; x < size; x++) {
    var condition = x == y;
    result += condition ? "#" : " ";
  }
  print(result);
}

/// ■ 패턴 3: X자 모양
/// - 양쪽 대각선 방향에 # 출력
void print1(int size, int y) {
  var result = "";
  for (var x = 0; x < size; x++) {
    var condition = (x == y || x == size - 1 - y);
    result += condition ? "#" : " ";
  }
  print(result);
}
```
